def show_a_random_face_per_class(images, unique_ids):
    #Creating 4X10 subplots in  18x9 figure size
    fig, axarr=plt.subplots(nrows=4, ncols=10, figsize=(18, 9))
    #For each iteration flattened 4X10 subplots matrix to 40 array
    axarr=axarr.flatten()

    #iterating over user ids
    rand=np.random.randint(10)
    for unique_id in unique_ids:
        image_index=unique_id*10+rand
        axarr[unique_id].imshow(images[image_index], cmap='gray')
        axarr[unique_id].set_xticks([])
        axarr[unique_id].set_yticks([])
        axarr[unique_id].set_title("class id:{}".format(unique_id))
    plt.suptitle("40 distinct people/classes in the dataset")

#(1) one-hidden layer NN with 10 hidden nodes
Acc1 = []
for train_index, test_index in kf.split(X):
    X_train, X_test = X[train_index], X[test_index]
    y_train, y_test = target[train_index], target[test_index]
    clf1 = MLPClassifier(solver='adam', hidden_layer_sizes=10,random_state=42, activation= 'logistic', max_iter=10000)
    clf1.fit(X_train, y_train)
    y_pred = clf1.predict(X_test)
    Acc1.append(accuracy_score(y_test, y_pred))
print("Accuracy of network with 10 hidden units:",Acc1,np.mean(Acc1),np.std(Acc1))
